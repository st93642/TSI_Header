#import <Foundation/Foundation.h>
#include <iostream>
#include <string>

/**
 * Main function - entry point of the program
 */
int main(int argc, const char * argv[]) {
    @autoreleasepool {
        // Objective-C style
        NSLog(@"Hello, World!");
        NSLog(@"This is a basic Objective-C++ program.");
        
        // C++ style
        std::cout << "C++ integration works!" << std::endl;
        
        // Mix of both
        std::string cppString = "TSI Student";
        NSString *objcString = [NSString stringWithUTF8String:cppString.c_str()];
        NSLog(@"Mixed languages: %@", objcString);
        
        // C++ STL containers
        std::vector<std::string> courses = {"Programming", "Mathematics", "Physics"};
        for (const auto& course : courses) {
            std::cout << "Course: " << course << std::endl;
        }
    }
    return 0;
}
