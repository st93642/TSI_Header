# Example function with type annotations
function greet(name::String)::String
    return "Hello, $name!"
end

# Example usage
# println(greet("TSI Student"))

# Julia supports multiple dispatch
function greet(name::String, enthusiasm::Int)
    exclamation = "!" ^ enthusiasm
    return "Hello, $name$exclamation"
end
