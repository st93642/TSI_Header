// Basic Objective-J program
// Note: Objective-J is used with Cappuccino framework

@import <Foundation/Foundation.j>
@import <AppKit/AppKit.j>

@implementation HelloWorld : CPObject
{
    CPString message;
}

// Main function - entry point
+ (void)main
{
    console.log("Hello, World!");
    console.log("This is a basic Objective-J program.");
    
    var app = [[HelloWorld alloc] init];
    [app greet:@"TSI Student"];
}

// Initialize
- (id)init
{
    self = [super init];
    if (self)
    {
        message = @"Hello from TSI!";
    }
    return self;
}

// Example method with parameter
- (void)greet:(CPString)name
{
    console.log("Hello, " + name + "!");
}

// Getter for message
- (CPString)message
{
    return message;
}

@end

// Execute main
[HelloWorld main];
