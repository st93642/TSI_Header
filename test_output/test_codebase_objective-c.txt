    @autoreleasepool {
        NSLog(@"Hello, World!");
        NSLog(@"This is a basic Objective-C program.");
        
        // Example with NSString
        NSString *message = @"Welcome to TSI!";
        NSLog(@"%@", message);
        
        // Example with NSArray
        NSArray *languages = @[@"C", @"Objective-C", @"Swift"];

}
