import React from 'react';
import ReactDOM from 'react-dom/client';

// Interface definitions
interface AppProps {}

interface AppState {
  message: string;
  count: number;
}

/**
 * Main React component
 */
function App(): JSX.Element {
  const [state, setState] = React.useState<AppState>({
    message: 'Hello, World!',
    count: 0
  });

  const handleIncrement = (): void => {
    setState(prev => ({ ...prev, count: prev.count + 1 }));
  };

  const handleMessageChange = (newMessage: string): void => {
    setState(prev => ({ ...prev, message: newMessage }));
  };

  return (
    <div className="app">
      <header className="app-header">
        <h1>{state.message}</h1>
        <p>This is a basic TypeScript React application.</p>
        <p>Count: {state.count}</p>
      </header>
      <main>
        <section>
          <h2>Welcome to TSI</h2>
          <p>Transport and Telecommunication Institute</p>
          <button onClick={handleIncrement}>
            Increment Counter
          </button>
          <button onClick={() => handleMessageChange('Hello, TSI Student!')}>
            Change Message
          </button>
        </section>
      </main>
    </div>
  );
}

/**
 * Render the application
 */
const root = ReactDOM.createRoot(
  document.getElementById('root') as HTMLElement
);
root.render(<App />);

export default App;
