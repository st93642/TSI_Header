" Basic Vimscript code
" Note: This is a Vim script for editor automation

" Main function - entry point
function! HelloWorld()
    echo "Hello, World!"
    echo "This is a basic Vimscript program."
endfunction

" Call main function
call HelloWorld()

" Example function with parameters
function! Greet(name)
    if a:name ==# ''
        let name = 'World'
    else
        let name = a:name
    endif
    return 'Hello, ' . name . '!'
endfunction
