<template>
  <div id="app" class="app">
    <header class="app-header">
      <h1>{{ message }}</h1>
      <p>This is a basic Vue.js application.</p>
      <p>Count: {{ count }}</p>
    </header>
    
    <main>
      <section>
        <h2>Welcome to TSI</h2>
        <p>Transport and Telecommunication Institute</p>
        
        <div class="buttons">
          <button @click="increment" class="btn">
            Increment Counter
          </button>
          <button @click="changeMessage" class="btn">
            Change Message
          </button>
          <button @click="reset" class="btn btn--secondary">
            Reset
          </button>
        </div>
        
        <div v-if="showWelcome" class="welcome">
          <p>Welcome, {{ studentName }}!</p>
        </div>
      </section>
    </main>
  </div>
</template>

<script>
export default {
  name: 'App',
  data() {
    return {
      message: 'Hello, World!',
      count: 0,
      studentName: 'TSI Student',
      showWelcome: false
    }
  },
  methods: {
    increment() {
      this.count++;
    },
    changeMessage() {
      this.message = 'Hello, TSI Student!';
      this.showWelcome = true;
    },
    reset() {
      this.message = 'Hello, World!';
      this.count = 0;
      this.showWelcome = false;
    }
  },
  mounted() {
    console.log('Vue app mounted successfully!');
  }
}
</script>
