; LOGO Program Example
; This demonstrates LOGO turtle graphics programming

; Clear the screen and reset turtle
CLEARSCREEN
HOME

; Set up the drawing environment
SETSCREENCOLOR [255 255 255]  ; White background
SETPC [0 0 0]                 ; Black pen color
SETPENSIZE [2]                 ; Medium pen size

; Define procedures for shapes

; Procedure to draw a square
TO SQUARE :size
  REPEAT 4 [FORWARD :size RIGHT 90]
END

; Procedure to draw a triangle
TO TRIANGLE :size
  REPEAT 3 [FORWARD :size RIGHT 120]
END

; Procedure to draw a circle (approximated with polygon)
TO CIRCLE :radius
  LOCAL "oldpos
  MAKE "oldpos POS
  PENDOWN
  REPEAT 36 [FORWARD (:radius * 3.14159 * 2 / 36) RIGHT 10]
  SETPOS :oldpos
END

; Procedure to draw a star
TO STAR :size
  REPEAT 5 [FORWARD :size RIGHT 144]
END

; Procedure to draw a flower
TO FLOWER :petals :size
  REPEAT :petals [
    CIRCLE :size
    RIGHT (360 / :petals)
  ]
END

; Main program demonstration

; Draw a colorful pattern
SETPC [255 0 0]    ; Red
SQUARE 50

PENUP
FORWARD 75
PENDOWN

SETPC [0 255 0]    ; Green
TRIANGLE 60

PENUP
FORWARD 75
RIGHT 90
FORWARD 50
LEFT 90
PENDOWN

SETPC [0 0 255]    ; Blue
STAR 40

; Draw a flower pattern
PENUP
SETPOS [-100 100]
PENDOWN

SETPC [255 0 255]  ; Magenta
FLOWER 6 20

; Demonstrate variables and arithmetic
MAKE "counter 0
MAKE "angle 0

; Draw a spiral pattern
PENUP
SETPOS [100 -100]
SETHEADING 0
PENDOWN

SETPC [255 165 0]  ; Orange
REPEAT 20 [
  FORWARD (:counter * 2)
  RIGHT 15
  MAKE "counter (:counter + 1)
]

; Demonstrate conditional logic
TO CHECK_SIZE :value
  IF :value > 100 [PRINT [BIG]]
  IF :value < 50 [PRINT [SMALL]]
  IF AND (:value >= 50) (:value <= 100) [PRINT [MEDIUM]]
END

; Test the conditional procedure
CHECK_SIZE 75
CHECK_SIZE 25
CHECK_SIZE 150

; Demonstrate loops with different patterns
PENUP
SETPOS [-150 0]
PENDOWN

SETPC [128 0 128]  ; Purple
REPEAT 8 [
  SQUARE 20
  PENUP
  FORWARD 30
  RIGHT 45
  PENDOWN
]

; Demonstrate text output
SETTEXTCOLOR [0 0 0]  ; Black text
LABEL [LOGO Graphics Demo Complete!]

; Demonstrate list operations
MAKE "colors [[255 0 0] [0 255 0] [0 0 255] [255 255 0]]
MAKE "shapes [SQUARE TRIANGLE CIRCLE STAR]

; Draw shapes with different colors
PENUP
SETPOS [0 0]
PENDOWN

FOR [i 1 4] [
  SETPC ITEM :i :colors
  RUN ITEM :i :shapes 30
  PENUP
  FORWARD 80
  PENDOWN
]

; Demonstrate recursion with a tree
TO TREE :size
  IF :size < 5 [STOP]
  FORWARD :size
  RIGHT 30
  TREE :size * 0.7
  LEFT 60
  TREE :size * 0.7
  RIGHT 30
  BACK :size
END

; Draw a recursive tree
PENUP
SETPOS [150 50]
SETHEADING 90
PENDOWN

SETPC [139 69 19]  ; Brown
TREE 60

; Final message
PENUP
SETPOS [0 -150]
SETTEXTCOLOR [255 0 0]
LABEL [TSI LOGO Programming Demo]

; Hide the turtle
HIDETURTLE

; Program completed
PRINT [LOGO program execution finished!]
