{ Basic Object Pascal/Delphi program }

program HelloWorld;

{$APPTYPE CONSOLE}

uses
  SysUtils;

type
  { Example class definition }
  TTSIStudent = class
  private
    FName: string;
    FProgram: string;
  public
    constructor Create(const AName, AProgram: string);
    procedure Introduce;
    property Name: string read FName;
    property Program: string read FProgram;
  end;

{ TTSIStudent implementation }
constructor TTSIStudent.Create(const AName, AProgram: string);
begin
  FName := AName;
  FProgram := AProgram;
end;

procedure TTSIStudent.Introduce;
begin
  Writeln('Hello, I''m ', FName, ' from TSI!');
end;

{ Main program }
var
  Student: TTSIStudent;
begin
  try
    Writeln('Hello, World!');
    Writeln('This is a basic Object Pascal program.');
    
    { Example usage }
    Student := TTSIStudent.Create('TSI Student', 'Computer Science');
    try
      Student.Introduce;
    finally
      Student.Free;
    end;
    
  except
    on E: Exception do
      Writeln(E.ClassName, ': ', E.Message);
  end;
end.
